{{- /* essay-count: returns the number of pages in a content section, pluralized as "(N Essay)" / "(N Essays)". Mirrors section-count. Counts regular pages only; section _index.md files are excluded. */ -}} {{- $section := .Get 0 -}} {{- $pages := where (where .Site.RegularPages "Section" $section) "Kind" "page" -}} {{- $count := len $pages -}} {{- if eq $count 1 -}} ({{ $count }} Essay) {{- else -}} ({{ $count }} Essays) {{- end -}}
